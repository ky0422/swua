extern print(int) -> int
extern print_str(str) -> str
extern to_str(int) -> str
extern concat_str(str, str) -> str

define fib(n int) -> int =
    if n < 2
        return n
    else
        return fib(n - 1) + fib(n - 2)

define main -> int =
    print_str(concat_str("fib(10) = ", to_str(fib(10))))

    let i = 0
    while i <= 10
        print(fib(i))
        i = i + 1

    return 0
